    import matplotlib.pyplot as plt

    import act

    time_window = [datetime(2020, 2, 4, 2, 0), datetime(2020, 2, 10, 10, 0)]
    station = 'KORD'
    my_asoses = act.discovery.get_asos(time_window, station='ORD')
